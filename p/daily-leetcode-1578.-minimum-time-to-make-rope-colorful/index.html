class Solution {
    public int minCost(String colors, int[] neededTime) {
        int left = 0;
        int right = 0;
        int totalTime = 0;
        char[] c = colors.toCharArray();
        while(right < c.length) {
            int currentGroupTime = 0;  // ç•¶å‰å…©å€‹é‡è¤‡é¡è‰²è¦ç§»é™¤çš„æ™‚é–“ç¸½å’Œ
